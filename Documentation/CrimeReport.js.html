import React from 'react';
import ReactDOM from 'react-dom';

import Paper from 'material-ui/Paper';

/**
 * Crime Report Component, 
 * This components get the crime data based on the location of the job.
 */
export default class CrimeReport extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            jobDetails: [],
		};
    }

    componentWillMount() {

		this.setState({
			jobDetails: this.props.JobDetails,
		});
    }


    render() {
      return(
        <div className="col-md-12 dashboard-component">
            <Paper className="CrimeDataComponent dashboard-paper" zDepth={1}
            data-toggle="tooltip" data-placement="bottom" title={`Crime report for this location`}>
                <h3>Crime Report for {this.state.jobDetails.formattedLocation} </h3>
                <h4>There are {this.props.crimeData.length} reported Crime Incidents in {this.state.jobDetails.formattedLocation} area.</h4>
                <h4>With {this.props.crimeCount.burglary} related to Burglary and {this.props.crimeCount.theft} to theft</h4>
            </Paper> 
        </div>
      )
    }
};
