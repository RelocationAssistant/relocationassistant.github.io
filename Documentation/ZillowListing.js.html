/**
 * House Listing Component, 
 * This component loads the details of property prices based on the Job location on the Dashboard.
*/
export default class HouseListing extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            jobDetails: [],
		};
    }

    componentWillMount() {

		this.setState({
			jobDetails: this.props.JobDetails,
		});
    }


    render() {
      return(
        <div className="col-md-12 dashboard-component">
            <Paper className="dashboard-houses-component dashboard-paper" zDepth={1}>

            <div id="zillow-large-search-box-widget-container" >

                <iframe scrolling="yes" src={`https://www.zillow.com/widgets/search/LargeSearchBoxWidget.htm?did=zillow-large-search-box-iframe-widget&type=iframe&rgname=${this.props.JobDetails.city}+${this.props.JobDetails.state}&shvi=yes`}  frameborder="0" ></iframe>
            
            </div>

            </Paper>   
        </div>
      )
    }
};
