import React, { Component } from 'react';

import { Redirect, Link } from 'react-router-dom';
import {browserHistory} from 'react-router';
import PropTypes from 'prop-types';

import MuiThemeProvider from 'material-ui/styles/MuiThemeProvider';
import getMuiTheme from 'material-ui/styles/getMuiTheme'
import MenuItem from 'material-ui/MenuItem';
import AppBar from 'material-ui/AppBar';
import Drawer from 'material-ui/Drawer';
import {lightBlue800} from 'material-ui/styles/colors';
import IconButton from 'material-ui/IconButton';
import NavigationClose from 'material-ui/svg-icons/navigation/close';
import MenuIcon from 'material-ui/svg-icons/navigation/menu';
import FlatButton from 'material-ui/FlatButton';
import LockIcon from 'material-ui/svg-icons/action/lock';

const muiTheme=getMuiTheme({
  palette: {
    primary1Color: lightBlue800,
  },
  appBar: {
    height: 70,
  },
});

/**
 * App Component (Main Parent Component), 
 * It is the parent component of the application.
 * It comtains the Menu and Drawer component 
 * and the rest of the components are loaded within this component
 */
export default class App extends Component {

  constructor() {
    super();
    this.state={
    };
  }

  handleToggle=() => this.setState({open: !this.state.open});
  handleClose=() => this.setState({open: false});


  static childContextTypes =
  {
      muiTheme: PropTypes.object
  }

  getChildContext()
  {
      return {
          muiTheme: getMuiTheme()
      }
  }

    componentDidMount() {
     
        this.attachTooltip();
    }

    componentDidUpdate() {
        this.attachTooltip();
    }

    attachTooltip = () => {
    // $(this.refs.data).tooltip();
    $('[title]').tooltip()    
    }



  render() {

    return (
      <div className="container">        
        <MuiThemeProvider muiTheme={getMuiTheme(muiTheme)}>
          <AppBar
            className="AppBar"
            iconElementLeft={<div>
              <FlatButton
                icon={<MenuIcon className="MenuIcon" />}
                onTouchTap={this.handleToggle}
                data-toggle="tooltip" data-placement="bottom" title="Menu"
              />
              <Link to="/">
                <FlatButton 
                    label="Relocation Assistant" 
                    className="AppBarTitle"
                />
              </Link>
              </div>
            }
                
            iconElementRight={
                <div>
                    <Link to="/login">
                        <IconButton data-toggle="tooltip" data-placement="left" title="Sign in"> <LockIcon /> </IconButton>
                    </Link>
                </div>
            }
          />
        </MuiThemeProvider>

        <MuiThemeProvider muiTheme={getMuiTheme(muiTheme)}>
          <Drawer
            className="MainDrawer"
            docked={false}
            width={250}
            open={this.state.open}
            onRequestChange={(open) => this.setState({open})}
          >
            <AppBar title="Menu"
              onLeftIconButtonTouchTap={this.handleToggle}
              iconElementLeft={<IconButton
                                data-toggle="tooltip" data-placement="bottom" title="Close Menu"
                                ><NavigationClose />
                               </IconButton>}
            />

            <Link to="/" >
              <MenuItem onTouchTap={this.handleClose}>Home</MenuItem>
            </Link>

            <Link to="/profile" >
              <MenuItem onTouchTap={this.handleClose}>Profile</MenuItem>
            </Link>

           

          </Drawer>
        </MuiThemeProvider>
      </div>
    );
  }
}
