import React from 'react';
import ReactDOM from 'react-dom';

import Paper from 'material-ui/Paper';

/**
 * Job Map Component, 
 * This component loads the map of the Job location on the Dashboard.
*/
export default class JobMap extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            mapIframeURL: "",
		};
    }

    componentWillMount() {

        this.setState({
            mapIframeURL: `//www.google.com/maps/embed/v1/place?q=${this.props.latitude},${this.props.longitude}&zoom=12&key=AIzaSyAVCVopLFl_D5S0SRdTJme7Cb8pjrdpPUE`,
		});
    }

    render() {
      return(
        <div className="col-md-12 dashboard-component">
            <Paper className="JobLocationMapComponent dashboard-paper"  zDepth={1}>
                <iframe src={this.state.mapIframeURL}
                width = "100%" height = "100%" frameBorder="0"
                >
                </iframe>
            </Paper>   
        </div>
      )
    }
};
